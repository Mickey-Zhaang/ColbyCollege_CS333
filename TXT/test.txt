#include <stdio.h>

// for txt files, use `gcc -xc -o program_name your_file.txt`

static int a = 0;

void mystery( int a ) {
	int b = 10;
	for (int a = 0; a < 3; a++) {
		b++;
	}
	a++;
	printf( "mystery b: %i\n", b );	

}

int main( int argc, char *argv[]) {
	
	printf( "main a: %i\n", a );
	mystery(a);
	printf( "main a: %i\n", a );
	mystery(a);
	printf( "main a: %i\n", a );

	
}
